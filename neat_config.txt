# NEAT Configuration for Direct Racing Controller
# Neural network receives 11 inputs and outputs steering_rate + acceleration
# Based on best practices from https://neat-python.readthedocs.io/en/latest/cookbook.html

[NEAT]
# Larger population for complex control problems
pop_size = 25
# Fitness criterion: max = maximize fitness (higher is better)
fitness_criterion = max
# High threshold - we use generation limit instead
fitness_threshold = 99999
# Reset on extinction to keep evolution going
reset_on_extinction = True
# No fitness termination - run for specified generations
no_fitness_termination = False

[DefaultGenome]
# Network topology
# 11 inputs: velocity, steering, lateral_error, heading_error, curvatures (3),
#            boundary distances (2), lookahead point (2)
num_inputs = 11
# 2 outputs: steering_rate, acceleration (both scaled from tanh [-1, 1])
num_outputs = 2
# Start with 2 hidden nodes for complex control
num_hidden = 2

# Node activation options
# Use tanh for outputs in range [-1, 1] which we scale to control ranges
activation_default = tanh
activation_mutate_rate = 0.0
activation_options = tanh

# Node aggregation options
aggregation_default = sum
aggregation_mutate_rate = 0.0
aggregation_options = sum

# Node bias options
bias_init_mean = 0.0
bias_init_stdev = 1.0
bias_init_type = gaussian
bias_max_value = 30.0
bias_min_value = -30.0
bias_mutate_power = 0.5
bias_mutate_rate = 0.7
bias_replace_rate = 0.1

# Genome compatibility options
compatibility_disjoint_coefficient = 1.0
compatibility_weight_coefficient = 0.5

# Connection add/remove rates - enable topology evolution
conn_add_prob = 0.5
conn_delete_prob = 0.5

# Connection enable options
enabled_default = True
enabled_mutate_rate = 0.01
enabled_rate_to_true_add = 0.0
enabled_rate_to_false_add = 0.0

# Recurrent network for control problems (memory of past states)
feed_forward = False

# Initial connection - partial without direct input-output connections
# Forces signal to flow through hidden nodes initially
initial_connection = partial_nodirect 0.5

# Connection weight options
weight_init_mean = 0.0
weight_init_stdev = 1.0
weight_init_type = gaussian
weight_max_value = 30
weight_min_value = -30
weight_mutate_power = 0.5
weight_mutate_rate = 0.8
weight_replace_rate = 0.1

# Node add/remove rates - enable topology evolution
node_add_prob = 0.2
node_delete_prob = 0.2

# Node response options
response_init_mean = 1.0
response_init_stdev = 0.0
response_init_type = gaussian
response_max_value = 30.0
response_min_value = -30.0
response_mutate_power = 0.0
response_mutate_rate = 0.0
response_replace_rate = 0.0

# Structural mutation inheritance
single_structural_mutation = False
structural_mutation_surer = default

[DefaultSpeciesSet]
# Compatibility threshold - lower = more species = more diversity
# Use 2.5 for complex problems (more exploration)
compatibility_threshold = 2.5

[DefaultStagnation]
# Species fitness function
species_fitness_func = max
# Max generations without improvement before species is removed
max_stagnation = 20
# Always keep at least this many species
species_elitism = 2

[DefaultReproduction]
# Number of top genomes to preserve unchanged each generation
elitism = 5
# Fraction of species allowed to reproduce
survival_threshold = 0.2
# Minimum number of genomes in a species
min_species_size = 2
